cmake_minimum_required(VERSION 3.14)

project(skipping_tests LANGUAGES CXX)

set(CMAKE_INCLUDE_CURRENT_DIR ON)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt5 REQUIRED COMPONENTS Core Test)

add_executable(skipping_tests
  main.cpp
)

add_executable(skipping_tests_test
  test.h
  test.cpp
)

target_link_libraries(skipping_tests Qt5::Core)

target_link_libraries(skipping_tests_test Qt5::Core Qt5::Test)
